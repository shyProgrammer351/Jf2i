프로그램 시작 
   점수들 입력 
   점수 정렬 
      영역의 마지막 칸에 가장 큰 값을 저장하면서, 영역을 한 칸씩 앞으로 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장

프로그램 코드 

	// 파일명 : ./Chapter02/ScoreSort.java 

	import java.lang.System; 

	import java.util.Scanner; 

	 

	public class ScoreSort 

	{ 

		// 프로그램 시작 

1		public static void main( String[] args )  

		{ 

			// 변수 초기화 

			Scanner scan = new Scanner( System.in ); 

2			int[] score = new int[ 3 ]; 

		 

			// 점수들 입력  

3			for( int index = 0; index < score.length; index  ++  )							 

			{ 

				System.out.print( "0점 ~ 100점 사이의 점수를 입력하세요: " ); 

				// 90, 89, 86을 순서대로 입력한다고 가정 

4				score[ index ] = scan.nextInt();  

			}  

 

			// 점수 정렬 

			// 영역의 마지막 칸에 가장 큰 값을 저장하면서, 영역을 한 칸씩 앞으로 좁혀가기 

5			for ( int max = score.length - 1; max >= 0; max --  )				 

			{ 

				// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

6				for ( int index = 0; index < max; index ++  )				 

				{ 

					// 앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 

7					if ( score[ index ] > score[ index + 1 ] ) 

					{ 

8						int temp = score[ index ]; 

9						score[ index ] = score[ index + 1 ];							 

10						score[ index + 1 ] = temp;								 

					} 

				}	 

			} 

	 

			// 정렬된 점수 출력 

11			for ( int index = 0; index < score.length; index ++  )							 

			{ 

12				System.out.println( ( index + 1 ) + ") " + score[ index ] + " " ); 

			}  

 

			// 프로그램 종료 

13			return; 

		} 

	}
