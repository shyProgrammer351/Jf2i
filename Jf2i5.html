알고리즘

프로그램 시작 
   점수들 입력 
   점수 정렬 
      영역의 마지막 칸에 가장 큰 값을 저장하면서, 영역을 한 칸씩 앞으로 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 
   정렬된 점수 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/scoresort.java 

	import java.lang.System; 

	import java.util.Scanner; 

	 

	public class scoresort 

	{ 

		// 프로그램 시작 

1		public static void main( String[] args )  

		{ 

			// 변수 초기화 

			Scanner scan = new Scanner( System.in ); 

2			int[] scores = new int[ 3 ]; 

		 

			// 점수들 입력  

3			for( int index = 0; index < scores.length; index  ++  )							 

			{ 

				System.out.print( "0점 ~ 100점 사이의 점수를 입력하세요: " ); 

				// 90, 89, 86을 순서대로 입력한다고 가정 

4				scores[ index ] = scan.nextInt();  

			}  

 

			// 점수 정렬 

			// 영역의 마지막 칸에 가장 큰 값을 저장하면서, 영역을 한 칸씩 앞으로 좁혀가기 

5			for ( int max = scores.length - 1; max >= 0; max --  )				 

			{ 

				// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

6				for ( int index = 0; index < max; index ++  )				 

				{ 

					// 앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 

7					if ( scores[ index ] > scores[ index + 1 ] ) 

					{ 

8						int temp = scores[ index ]; 

9						scores[ index ] = scores[ index + 1 ];							 

10						scores[ index + 1 ] = temp;								 

					} 

				}	 

			} 

	 

			// 정렬된 점수 출력 

11			for ( int index = 0; index < scores.length; index ++  )							 

			{ 

12				System.out.println( scores[ index ] + " " ); 

			}  

 

			// 프로그램 종료 

13			return; 

		} 

	}
