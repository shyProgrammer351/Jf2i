프로그램 시작 
   손님에게 주문 받기 
      메인요리 주문 받기 
      소스 주문 받기

프로그램 코드 

	// 파일명 : ./Chapter02/FoodOrder.java

	import java.util.Scanner;

		 

	public class FoodOrder

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			// 변수 초기화

			Scanner scan = new Scanner( System.in );

2			String mainDish = "", sauce = "", drink = "";

		 

			// 손님에게 주문 받기 

3			System.out.println( "주문 도와드리겠습니다, 손님." );

		 

			// 메인요리 주문 받기 

4			System.out.print( "메인요리는 고르셨습니까: " );

5			mainDish = scan.nextLine().trim();

		 

			// 소스 주문 받기 

6			System.out.print( "소스는 어떻게 할까요: " );

7			sauce = scan.nextLine().trim();

		 

			// 음료 주문 받기 

8			System.out.print( "음료는 무엇으로 준비할까요: " );

9			drink = scan.nextLine().trim();

		 

			// 주문 내역 확인 

10			System.out.println( "메인요리는 " + mainDish + ", 소스는 " + sauce + ", 음료는 " + drink + "를 선택하셨습니다. " );

11			System.out.println( "준비하겠습니다. 잠시만 기다려 주십시오." );

		 

		// 프로그램 종료 

12		}

	}
