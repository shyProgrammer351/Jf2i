6			month = scan.nextInt();

7			System.out.print( "당신이 태어난 날을 입력하세요: " );

8			day = scan.nextInt();

9			System.out.println( "당신은 " + year + "년 " + month + "월 " + day + "일에 태어났군요. 당신은 ♪~ 사랑받기 위해 ♪~ 태어난 사람 ♬~~ " );
