알고리즘

프로그램 시작 
   친구와 대화하기 
      이름 물어보고 답하기 
      직업 물어보고 답하기 
      취미 물어보고 답하기 
      좋아하는 색 물어보고 답하기 
      친하게 지내자고 인사하기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter02/Chat.java

	import java.util.Scanner;

			 

	public class Chat

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			Scanner scan = new Scanner( System.in );

			// 친구와 대화하기 

			// 이름 물어보고 답하기 

2			System.out.print( "안녕? 넌 이름이 뭐야: " );

3			String name = scan.next();

4			System.out.println( "만나서 반가워, " + name + ". " );

		 

			// 직업 물어보고 답하기 

5			System.out.print( "넌 주로 뭐하고 살아: " );

6			String job = scan.next();

7			System.out.println( job + "? 나도 마찬가지야 ㅎㅎ " );

		 

			// 취미 물어보고 답하기 

8			System.out.print( "그럼 심심할 때 뭐하니: " );

9			String hobby = scan.next();

10			System.out.println( hobby + "? 나랑 똑같네. " );

		 

			// 좋아하는 색 물어보고 답하기 

11			System.out.print( "혹시 좋아하는 색은 뭐야 : " );

12			String color = scan.next();

13			System.out.println( color + "이라구? 와! 이것도 나랑 똑같네. " );

		 

			// 친하게 지내자고 인사하기 

14			System.out.println( "앞으로 우리 친하게 지내자." );

			scan.close();

		// 프로그램 종료 

15		}

	}
