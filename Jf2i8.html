	// 파일명 : ./Chapter02/Chat.java 

	import java.lang.System; 

	import java.util.Scanner; 

	 

	public class Chat 

	{ 

		// 프로그램 시작 

1		public static void main( String[] args )  

		{ 

			// 변수 초기화

			String[][] dialogue =   							 

				{ { "이름이 뭐", "난 재석이야. 국민MC 유재석과 이름이 같아. ㅎㅎㅎ 넌 몇 살이니?" }, 

				  { "몇 살", "난 20살이야. 남들이 다 부러워하는 나이지. 넌 어디 살아?" },

3			while( true )
